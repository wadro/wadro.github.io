func pow(x, n, lim float64) float64 {
	if v := math.Pow(x, n); v < lim { // if 변수 선언(스코프는 if 또는 else {}블록안에서만 유효함)
		return v
	}
	return lim
}

func adder() func(int) int {
	var sum = 0
	return func(x int) int {
		sum += x
		return sum
	}
}

func main() {
	pos, neg := adder(), adder()
	for i := 0; i < 10; i++ {
		fmt.Println(
			pos(i),
			neg(-2*i),
		)
	}
}
